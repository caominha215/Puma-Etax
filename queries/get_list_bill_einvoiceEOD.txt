SELECT 
    d.SID,
    COALESCE(d.REF_SALE_SID, '') AS REF_SALE_SID,
    d.WORKSTATION_NAME
FROM document d
WHERE  status = 4 and DATE_FORMAT(d.INVC_POST_DATE, '%Y-%m-%d') = DATE_FORMAT(CURRENT_DATE(), '%Y-%m-%d')
AND NOT EXISTS (
    SELECT 1
    FROM EINVOICE_HISTORY e
    WHERE FIND_IN_SET(CAST(d.SID AS CHAR), REPLACE(e.SID, '/', ',')) > 0
);
